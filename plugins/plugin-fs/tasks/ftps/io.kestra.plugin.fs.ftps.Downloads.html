    Downloads 

    type: "io.kestra.plugin.fs.ftps.Downloads"

    Download multiple files from FTPS server

    # Examples

    Download a list of files and move it to an archive folders

    id: "downloads"
type: "io.kestra.plugin.fs.ftps.Downloads"
host: localhost
port: 990
username: foo
password: pass
from: "/in/"
interval: PT10S
action: MOVE
moveDirectory: "/archive/"
